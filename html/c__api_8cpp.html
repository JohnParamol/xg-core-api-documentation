◆ XGCoreGetNodeCount()

XG_API int XGCoreGetNodeCount ( HNodeManager  in_manager)

Get the number of Node objects.

Node objects must be previously added by XGCoreAddNodeFromClassIndex() or XGCoreAddNodeFromClassName(). Note that there is always +1 Node object - the master output node (see XGCoreGetMasterOutputNodeId()), so the function never returns a value less than 1.

Example code

xg_core::HNodeManager node_manager = XGCoreNodeManagerCreate();
int node_class_count = XGCoreGetNodeClassCount(node_manager);
for (int i = 0; i < node_class_count; ++i)
{
int node_class_name_len = XGCoreGetNodeClassNameByClassIndex(node_manager, i, nullptr);
std::unique_ptr<char[]> node_class_name(new char[node_class_name_len]);
XGCoreGetNodeClassNameByClassIndex(node_manager, i, node_class_name.get());
XGCoreAddNodeFromClassName(node_manager, node_class_name.get(), true);
}
int node_count = XGCoreGetNodeCount(node_manager);
for (int i = 0; i < node_count; ++i)
{
uint32_t node_id = XGCoreGetNodeIdByIndex(node_manager, i);
int node_class_name_len = XGCoreGetNodeClassNameById(node_manager, node_id, nullptr);
std::unique_ptr<char[]> node_class_name(new char[node_class_name_len]);
XGCoreGetNodeClassNameById(node_manager, node_id, node_class_name.get());
std::string s = "node #" + std::to_string(i + 1) + " (" + std::string(node_class_name.get()) + ")";
puts(s.c_str());
xg_core::HNodeClassInfo info = XGCoreGetNodeClassInfo(node_manager, node_id);
if (info)
{
int description_len = XGCoreGetNodeClassDescription(node_manager, info, nullptr);
std::unique_ptr<char[]> description(new char[description_len]);
XGCoreGetNodeClassDescription(node_manager, info, description.get());
puts(description.get());
}
}
