Reset a 3D LUT object to initial state (reset any color transformation).

Parameters
inout_lut3D LUT object handle
in_min_valueMinimum value for LUT data buffer (default is 0.f)
in_max_valueMaximum value for LUT data buffer (default is 1.f)
in_channelIndex of a single channel to reset (specify -1 to reset all channels)

Example code

const int image_buffer_size = 4;
xg_core::TColor image_buffer[image_buffer_size] = { {0.6, 0.7, 0.8}, {0.1, 0.2, 0.3}, {1, 0.5, 0}, {-1, -0.5, 1.5} };
xg_core::HNodeManager node_manager = XGCoreNodeManagerCreate();
int node_class_count = XGCoreGetNodeClassCount(node_manager);
for (int i = 0; i < node_class_count; ++i)
{
for (int repeat = 0; repeat < 2; ++repeat)
{
XGCoreAddNodeFromClassIndex(node_manager, i, true);
}
}
xg_core::HLut3D lut = XGCoreLut3DCreate(25, 3);
if (lut)
{
// Reset any color transformation for the LUT (all channels)
XGCoreLut3DReset(lut, 0.f, 1.f, -1);
int random_node_index = (rand() % XGCoreGetNodeCount(node_manager));
xg_core::TNodeId from_node_id = XGCoreGetNodeIdByIndex(node_manager, random_node_index);
// Merge the pipeline color transformation from a random
// node (not including it) to the last node in the LUT.
// In this case, the very first node will always be ignored.
// To include it in the process, you need to set from_node_id to 0.
XGCorePipelineMergeToLut3D(node_manager, lut, from_node_id, 0);
// Apply the transformed LUT to the buffer
XGCoreLut3DProcessRgbBuffer(lut, (float*)image_buffer, image_buffer_size);
XGCoreLut3DDelete(lut);
}
