{
xg_core::THandleNodeClassDescription desc = XGCoreNodeManagerGetNodeClassDescription(node_manager, i);
if (desc)
{
int class_name_len = XGCoreNodeManagerGetNodeClassName(node_manager, i, nullptr);
std::unique_ptr<char[]> class_name(new char[class_name_len]);
XGCoreNodeManagerGetNodeClassName(node_manager, i, class_name.get());
int description_len = XGCoreNodeClassDescriptionGetDescription(desc, nullptr);
std::unique_ptr<char[]> description(new char[description_len]);
XGCoreNodeClassDescriptionGetDescription(desc, description.get());
xg_core::EImageProcessComplexity complexity = XGCoreNodeClassDescriptionGetImageProcessMaxComplexity(desc);
std::string message = std::string(class_name.get()) + " (" + std::string(description.get()) +
"), complexity: " + std::string(xg_core::EImageProcessComplexity_ToDescription(complexity));
puts(message.c_str());
}
}
XGCoreNodeManagerDelete(&node_manager);
